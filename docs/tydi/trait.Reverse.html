Implementing Reverse

use tydi::{Reverse, Reversed};

#[derive(Clone, Copy, Debug, PartialEq)]
struct A {
    in_port: bool,
    size: u32,
}

impl Reverse for A {
    fn reverse(&mut self) {
        self.in_port = !self.in_port;
    }
}

let mut a = A { in_port: false, size: 42 };
a.reverse();
assert!(a.in_port);
a.reverse();
assert!(!a.in_port);
let b = A { in_port: true, size: 42 };
assert_eq!(a.reversed(), b);
